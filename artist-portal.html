import { _supabase } from './supabase-config.js'; const handleUpload = async (e) => { e.preventDefault(); const btn = document.getElementById('submitBtn'); const audioFile = document.getElementById('audioFile').files[0]; const coverFile = document.getElementById('coverFile').files[0]; const plan = localStorage.getItem('selectedPlan') || 'Go Free'; btn.disabled = true; btn.innerText = "LOCKING INTO VAULT..."; try { // 1. Upload Files to Storage const audioPath = `releases/audio/${Date.now()}_${audioFile.name}`; const coverPath = `releases/covers/${Date.now()}_${coverFile.name}`; await _supabase.storage.from('vault-assets').upload(audioPath, audioFile); await _supabase.storage.from('vault-assets').upload(coverPath, coverFile); // 2. Save Metadata for RouteNote const { error: dbErr } = await _supabase.from('releases').insert([{ artist_name: document.getElementById('artistName').value, track_title: document.getElementById('releaseTitle').value, genre: document.getElementById('genre').value, plan_type: plan, audio_url: audioPath, cover_url: coverPath, status: 'Reviewing' }]); if (dbErr) throw dbErr; window.location.href = 'upload-success.html'; } catch (err) { alert("Upload Failed: " + err.message); btn.disabled = false; } };
